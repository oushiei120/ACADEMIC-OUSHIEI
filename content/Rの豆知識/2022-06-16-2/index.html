本节会讲正后发断言,负后发断言

library(stringi)
library(tidyverse)
library(jiebaR)
x3 <- c("我喜欢你啊,我不喜欢你呢")
seg <- worker()
v1 <- segment(x3,seg)
v2 <- paste0(v1, collapse = " ");v2
## [1] "我 喜欢 你 啊 我 不 喜欢 你 呢"
str_extract_all(v2,"(?>=我).+?\\s")##没太搞懂
## [[1]]
## character(0)
str_extract_all(v2,"(?<!我\\s)(喜欢)")
## [[1]]
## [1] "喜欢"
